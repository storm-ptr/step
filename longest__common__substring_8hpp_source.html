35  }
36  }
37  }

42 template <template <class...> class Map>
43 struct suffix_tree_searcher {
44  template <class Size, class RandomIt1, class RandomIt2>
45  auto find_with(std::pair<RandomIt1, RandomIt1> rng1,
46  std::pair<RandomIt2, RandomIt2> rng2) const
47  {
48  auto result = std::make_pair(rng1.second, rng1.second);
49  auto tree = suffix_tree<iter_value_t<RandomIt1>, Size, Map>{};
50  append(tree, rng1, rng2);
51  auto flags = std::unordered_map<Size, uint8_t>{};
52  auto size1 = (Size)size(rng1);
53  tree.visit([&](auto& edge) {
54  if (!tree.leaf(edge.child))
55  flags[edge.parent] |= flags[edge.child];
56  else if (tree.path(edge).first < size1)
57  flags[edge.parent] |= 1;
58  else
59  flags[edge.parent] |= 2;
60  });
61  tree.visit([&](auto& edge) {
62  if (edge.visited && flags[edge.child] == (1 | 2) &&
63  edge.path > (Size)size(result)) {
64  auto [first, last] = tree.path(edge);
65  result.first = rng1.first + first;
66  result.second = rng1.first + last;
67  }
68  });
69  return result;
70  }
