12 template <class Compare>
13 struct suffix_array_searcher {
14  template <class Size, class RandomIt>
15  auto find_with(std::pair<RandomIt, RandomIt> rng) const
16  {
17  using value_t = iter_value_t<RandomIt>;
18  auto result = std::make_pair(rng.second, rng.second);
19  auto arr = suffix_array<value_t, Size, Compare>{rng.first, rng.second};
20  auto lcp = std::vector<Size>(arr.size());
21  arr.longest_common_prefix_array(lcp.begin());
22  auto it = std::max_element(lcp.begin(), lcp.end());
23  if (it != lcp.end() && *it > 0) {
24  auto pos = arr.nth_element((Size)std::distance(lcp.begin(), it));
25  result.first = rng.first + pos;
26  result.second = result.first + *it;
27  }
28  return result;
29  }
30 };

32 template <template <class...> class Map>
33 struct suffix_tree_searcher {
34  template <class Size, class RandomIt>
35  auto find_with(std::pair<RandomIt, RandomIt> rng) const
36  {
37  auto result = std::make_pair(rng.second, rng.second);
38  auto tree = suffix_tree<iter_value_t<RandomIt>, Size, Map>{};
39  append(tree, rng);
40  tree.visit([&](auto& edge) {
41  if (edge.visited && edge.path > (Size)size(result)) {
42  auto [first, last] = tree.path(edge);
43  result.first = rng.first + first;
44  result.second = rng.first + last;
45  }
46  });
47  return result;
48  }
